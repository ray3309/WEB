CREATE TABLE board(
no number(8) CONSTRAINT board_no_pk PRIMARY KEY,
title VARCHAR2(100) NOT NULL,
writer VARCHAR2(50),
content VARCHAR2(400),
pwd VARCHAR2(10),
hit NUMBER,
reip VARCHAR2(50),
regdate date
);
CREATE SEQUENCE board_seq 
INCREMENT BY 1 START WITH 1;
